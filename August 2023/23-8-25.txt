Problem Statement: 
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------

Palindrome String

-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
Solution: 
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
class Solution{
public:	
	
	
	int isPalindrome(string S)
	{
	    // Your code goes here
	    int bFlag = 1;
	    int length = S.length();
	    int i=0;
	    while(i<length-1){
	        if(S[i] != S[length-1]){
	            bFlag = 0;
	            break;
	        }
	        i++;
	        length--;
	    }
	    return bFlag;
	}

};
