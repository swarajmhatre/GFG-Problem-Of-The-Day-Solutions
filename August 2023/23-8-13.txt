nth Fibonacci Number

//Solution_1
class Solution {
  public:
    int nthFibonacci(int n){
        // code here
        int MOD = 1000000007;
        if(n ==1){
            return 1;
        }
        else if(n == 2){
            return 1;
        }
        else{
            int a =1,b =1;
            for(int i=2; i<n; i++){
                int temp =b;
                b = (a+b)%MOD;
                a =temp;
            }
            return b;
        }
        
    }
};

class Solution {

        // code here
        if(n ==1){
            return 1;
        }
        else if(n == 2){
            return 1;
        }
        else{
