Problem Statement: 
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------

Subarray with Given Sum

-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
Solution: 
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------

---------------------------------------------------------------------
class Solution{
    //Function to find the leaders in the array.
    public:
     vector<int> subarraySum(vector<int>arr, int n, long long s)
    {
        // Your code here
        if(s==0L)
        {
            return {-1};
        }
        vector<int>v;
        int i=0;
        int j=1;
        long long sum=0;
        sum=arr[i]+arr[j];
        while(i<n && j<n)
        {
            if(arr[i]==s)
            {
                return {i+1,i+1};
            }
            if(sum==s)
            {
                v.push_back(i+1);
                 v.push_back(j+1);
                 return v;
            }
            if(sum<s)
            {
                j++;
                sum+=arr[j];
            }
            else
            {
                sum-=arr[i];
                i++;
                 if(sum==s)
            {
                v.push_back(i+1);
                 v.push_back(j+1);
                 return v;
            }
                
            }
        }
        v.push_back(-1);
        return v;
    }
};
