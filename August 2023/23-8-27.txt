//User function Template for C++
class Solution
{
    public:
    string reverseWord(string str)
    {
        // Your code goes here
        int i =0;
        int length = str.length();
        while(i<length-1){
            swap(str[i], str[length-1]);
            i++;
            length--;
        }
        return str;
    }
};

	while(t--)

    	string s;

    	cout << ob.reverseWord(s) << endl;
	}
	return 0;
