-----------------------------------------------------------------------------------------------------------------------------------------------------------------------	
Problem Statement: 
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------

You are given a stack St. You have to reverse the stack using recursion.

-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
Solution: 
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------

class Solution{
public:
    void fix(stack<int> &St, queue<int> &q){
        if(St.empty()){
            return;
        }    
        
        q.push(St.top());
        St.pop();
        fix(St, q);
        St.push(q.front());
        q.pop();
    }
    
    void Reverse(stack<int> &St){
        queue<int> q;
        
        fix(St, q);
    }
};
