-----------------------------------------------------------------------------------------------------------------------------------------------------------------------	
Problem Statement: 
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------

Non repeating numbers

-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
Solution: 
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
class Solution
{
public:
    vector<int> singleNumber(vector<int> nums) 
    {
        // Code here.
        int value =0;
        for(auto x: nums){
            value ^= x;
        }
        
        int rightmost = (value & (value-1))^value;
        
        vector<int> ans(2,0);
        
        for(auto x : nums){
            if(x & rightmost){
                ans[0] ^= x;
            }
            else{
                ans[1] ^= x;
            }
        }
        
        sort(ans.begin(), ans.end());
        
        return ans;
    }
};
