Problem Statement: 
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
Given a number and its reverse. Find that number raised to the power of its own reverse.
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
Solution: 
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
class Solution{
    public:
    //You need to complete this fucntion
    
    long long power(int N,int R)
    {
       //Your code here
        int mod = 1000000007;
        if(R==1){
            return N%mod;
        }
         
        long long llHalf = power(N,R/2);
        llHalf = llHalf*llHalf%mod;
        if(R%2==1){
             llHalf = (llHalf*N)%mod;
        }
        return llHalf;
    }

};
