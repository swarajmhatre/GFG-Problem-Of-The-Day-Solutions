Problem Statement: 
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
Given a binary tree and the task is to find the spiral order traversal of the tree.
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
Solution: 
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------

vector<int> findSpiral(Node *root)
{
    //Your code here
    vector <int> ans;
    queue<Node *> Q;
    Q.push(root);
    bool flag = true;
    
    while(Q.size()){
        vector <int> seq;
        int s = Q.size();
        
        for(int i=0; i< s; i++){
            Node * temp = Q.front();
            Q.pop();
            seq.push_back(temp->data);
            if(temp->left != NULL)Q.push(temp->left);
            if(temp->right != NULL)Q.push(temp->right);
        }
        
        
        if(flag==true)reverse(seq.begin(), seq.end());
        for(auto item: seq){
            ans.push_back(item);
        }
        flag = !flag;
    }
    return ans;
}
