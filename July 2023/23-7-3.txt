Problem Statement: 
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------

Given an array arr[] of n positive integers. The task is to find the maximum of j - i subjected to the constraint of arr[i] <= arr[j].

-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
Solution: 
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
class Solution{
public:
    int maxIndexDiff(int arr[], int n) {
        // code here
    int i=0;
    int j =n-1;
    int iRet =0;
    while(i<j){
        if((n-i)<iRet){
            break;
        }
        if(arr[i]<=arr[j]){
             iRet = max(iRet, j-i);
             i++;
             j = n-1;
        }
      else{
          j--;
      }
    }
    return iRet;
    }
    

};
