Problem Statement: 
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
Given a Queue Q containing N elements. The task is to reverse the Queue. Your task is to complete the function rev(), that reverses the N elements of the queue.
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
Solution: 
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
class Solution
{
    public:
    queue<int> rev(queue<int> q)
    {
        
        stack<int> st;
        while(q.size()){
            st.push(q.front());
            q.pop();
        }
        while(st.size()){
            q.push(st.top());
            st.pop();
        }
        return q;

    }
};
